Find the Smaller Angle

        
var Minimal_Angle = (h, m) => 
{
    if(h===12){
      h=0;
    }
    if(m===60){
      m=0;
      h+=1;
      if(h < 12){
        h =h-12
      }
    }
    let hAngle = 0.5*(h*60+m)
    let mAngle = 6*m;
    let result = Math.abs(hAngle-mAngle)
    result = Math.min((360-result),result)
    return result
};

Perfect Number Check

 
var Perfect_Check = (N) => 
{
     if(N===1){
    return "YES"
  }else{
    let sum = 0;
    for(let i = 1;i< N;i++){
      if(N%i===0){
        sum += i
      }
    }
    if (sum === N){
      return "YES"
    }else{
      return "NO"
    }
  }
};
