Count the Heads

var Count_Occ = (s) => 
{
  let arr = [];
  let arr2 = s.split('');
  let str = "";
  let count = 1;
  for(let i = 0; i < arr2.length-1; i++){
    for(let j = i+1; j < arr2.length; j++){
      if(arr2[i] === arr2[j]){
        count++;
        arr2.splice(j,1);
        j--;
      }
    }
    if(count > 1){
      str = `${arr2[i]}${count}`;
      arr.push(str);
      count = 1;
    }
  }
  let sorted = arr.sort();
  return sorted.join('');
};

{

Reverse the String

    
var Reverse_String = (str) => 
{
     
    var newString = "";
    for (var i = str.length - 1; i >= 0; i--) {
        newString += str[i];
    }
    return newString;
    
};

var Count_Vowels= (S) => 
{
    let count = 0;
    let Vowels = ["a","e","i","o","u"];

    for(let letter of S.toLowerCase()){
        for(let V of Vowels){
            if(letter===V){
                count++;
            }
        }
    }
    return count;
};
var Count_Consonants= (S) => 
{
  return S.length - Count_Vowels(S);
};
